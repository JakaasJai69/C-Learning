28 // Visual Studio is non-conforming in support for __cplusplus (unless you set a specific compiler flag, which you probably haven't)
29 // In Visual Studio 2015 or newer we can use _MSVC_LANG instead
30 // See https://devblogs.microsoft.com/cppblog/msvc-now-correctly-reports-__cplusplus/
31#if defined(_MSVC_LANG)
32 return _MSVC_LANG;
33#elif defined(_MSC_VER)
34 // If we're using an older version of Visual Studio, bail out
35 return -1;
36#else
37 // __cplusplus is the intended way to query the language standard code (as defined by the language standards)
38 return __cplusplus;

86 if (standard == -1)

93 for (int i = 0; i < numStandards; ++i)
94 {
95 // If the reported version is one of the finalized standard codes
96 // then we know exactly what version the compiler is running
97 if (standard == stdCode[i])

99 std::cout << "Your compiler is using " << stdName[i]

101 << "L)\n";
102 break;
103 }
104
105 // If the reported version is between two finalized standard codes,
106 // this must be a preview / experimental support for the next upcoming version.
107 if (standard < stdCode[i])
108 {
109 std::cout << "Your compiler is using a "
110 "preview/pre-release of "

112 << standard << "L)\n";
